FUNCTION grouped_split(dataframe, holdout_fraction=0.20, seed=42):
    /*
    Ensures no passage leakage between train/val/test
    All questions from same passage stay in same split
    */
    
    // First split: Train vs (Val + Test)
    groups = dataframe["passage_id"]
    train_indices, holdout_indices = group_shuffle_split(
        data=dataframe,
        groups=groups,
        test_size=holdout_fraction,
        random_state=seed
    )
    
    train_data = dataframe[train_indices]
    holdout_data = dataframe[holdout_indices]
    
    // Second split: Val vs Test (50-50 from holdout)
    holdout_groups = holdout_data["passage_id"]
    val_indices, test_indices = group_shuffle_split(
        data=holdout_data,
        groups=holdout_groups,
        test_size=0.5,
        random_state=seed
    )
    
    val_data = holdout_data[val_indices]
    test_data = holdout_data[test_indices]
    
    RETURN train_data, val_data, test_data
END FUNCTION
