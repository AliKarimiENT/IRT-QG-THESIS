FUNCTION main():
    // Configuration
    config = {
        "data_path": "/path/to/data.json",
        "base_model": "google/flan-t5-base",
        "output_dir": "./ielts_qg_model",
        
        // Training
        "epochs": 15,
        "batch_size": 4,
        "gradient_accumulation_steps": 4,
        "lr": 8e-5,
        "warmup_ratio": 0.15,
        "weight_decay": 0.01,
        "label_smoothing": 0.1,
        "patience": 7,
        
        // Sequence lengths
        "max_source_len": 512,
        "max_target_len": 256,
        
        // Generation
        "num_beams": 4,
        "length_penalty": 1.0,
        
        // Optimization
        "fp16": True,
        "grad_ckpt": False,
        
        // LoRA
        "use_lora": True,
        "lora_r": 16,
        "lora_alpha": 32,
        "lora_dropout": 0.05,
        
        "seed": 42
    }
    
    // 1. Seed for reproducibility
    set_seed(config["seed"])
    
    // 2. Load and prepare data
    PRINT "Loading data..."
    data = prepare_dataset(config["data_path"], config)
    
    PRINT "Extracting control tokens..."
    control_tokens = extract_type_tokens(data)
    
    PRINT "Splitting data..."
    train_df, val_df, test_df = grouped_split(data, holdout_fraction=0.20)
    
    // 3. Initialize model
    PRINT "Preparing model..."
    model, tokenizer = prepare_model(
        config["base_model"], 
        config, 
        control_tokens
    )
    
    // 4. Create datasets
    PRINT "Building datasets..."
    train_dataset = QGDataset(
        train_df, tokenizer, 
        config["max_source_len"], 
        config["max_target_len"]
    )
    
    val_dataset = QGDataset(
        val_df, tokenizer,
        config["max_source_len"],
        config["max_target_len"]
    )
    
    // 5. Train
    PRINT "Training model..."
    trainer, results = train_model(
        model, tokenizer,
        train_dataset, val_dataset,
        config
    )
    
    // 6. Evaluate on test set
    IF length(test_df) > 0:
        PRINT "Evaluating on test set..."
        test_dataset = QGDataset(
            test_df, tokenizer,
            config["max_source_len"],
            config["max_target_len"]
        )
        test_metrics = trainer.evaluate(eval_dataset=test_dataset)
        PRINT test_metrics
    
    // 7. Demo generation
    PRINT "Running demo generation..."
    demo_cases = [
        ("Easy", "readingShortAnswer", sample_passage_1),
        ("Medium", "readingTrueFalseNotGiven", sample_passage_2),
        ("Hard", "readingMultipleChoices", sample_passage_3)
    ]
    
    FOR difficulty, qtype, passage in demo_cases:
        generated = generate_question(
            model, tokenizer, passage, difficulty, qtype, config
        )
        PRINT f"Difficulty: {difficulty}, Type: {qtype}"
        PRINT f"Generated: {generated}"
    
    PRINT "Training complete!"
END FUNCTION
