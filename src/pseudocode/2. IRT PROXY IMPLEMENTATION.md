// Item Response Theory Proxy Mapping
CONSTANT IRT_PROXY_MAP = {
    "Easy": -1.0,      // Lower difficulty parameter
    "Medium": 0.0,     // Neutral difficulty  
    "Hard": 1.0        // Higher difficulty parameter
}

CONSTANT IRT_DEFAULTS = {
    "discrimination": 1.0,  // Parameter 'a' - item discrimination
    "guessing": 0.0        // Parameter 'c' - pseudo-guessing
}

FUNCTION get_irt_parameters(difficulty_level):
    /*
    Maps categorical difficulty to IRT 'b' parameter (item difficulty)
    In IRT: P(θ) = c + (1-c) / (1 + exp(-a(θ - b)))
    where:
        - θ (theta): ability level
        - a: discrimination 
        - b: difficulty (what we control)
        - c: guessing probability
    */
    
    b_value = IRT_PROXY_MAP.get(difficulty_level, 0.0)
    a_value = IRT_DEFAULTS["discrimination"]
    c_value = IRT_DEFAULTS["guessing"]
    
    RETURN (a_value, b_value, c_value)
END FUNCTION
