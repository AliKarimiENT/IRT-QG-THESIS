FUNCTION prepare_model(base_model_name, config, special_tokens):
    // Load base model
    tokenizer = T5TokenizerFast.from_pretrained(base_model_name)
    model = T5ForConditionalGeneration.from_pretrained(base_model_name)
    
    // Add special tokens
    new_tokens = ["<SEP>", "<BLANK>"] + special_tokens  // Type tokens
    num_added = tokenizer.add_special_tokens({
        "additional_special_tokens": new_tokens
    })
    
    IF num_added > 0:
        model.resize_token_embeddings(len(tokenizer))
        PRINT f"Added {num_added} special tokens"
    
    // Optional: Enable gradient checkpointing
    IF config["grad_ckpt"]:
        model.gradient_checkpointing_enable()
    
    // Optional: Apply LoRA for parameter-efficient fine-tuning
    IF config["use_lora"]:
        lora_config = {
            "task_type": "SEQ_2_SEQ_LM",
            "r": config["lora_r"],                    // Rank (16)
            "lora_alpha": config["lora_alpha"],        // Scaling (32)
            "lora_dropout": config["lora_dropout"],    // Dropout (0.05)
            "target_modules": ["q", "k", "v", "o", "wi_0", "wi_1", "wo"]
        }
        model = get_peft_model(model, lora_config)
        model.print_trainable_parameters()
    
    // Set generation defaults
    model.config.max_length = config["max_target_len"]
    model.config.num_beams = config["num_beams"]
    model.config.length_penalty = config["length_penalty"]
    model.config.early_stopping = True
    
    RETURN model, tokenizer
END FUNCTION
