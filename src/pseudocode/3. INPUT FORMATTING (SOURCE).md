FUNCTION format_source_input(passage, difficulty, question_type, use_prefix=True):
    /*
    Creates model input with IRT parameters as control signals
    Format: [IRT_PARAMS] [PREFIX] [TYPE_TOKEN] Passage: {text}
    */
    
    // Step 1: Get IRT parameters
    (a, b, c) = get_irt_parameters(difficulty)
    irt_string = f"[a={a}, b={b}, c={c}]"
    
    // Step 2: Get question type-specific instruction prefix
    IF use_prefix:
        prefix = get_task_prefix(question_type)
        // Examples:
        // "TRUE/FALSE/NOT GIVEN" -> "Generate a TRUE/FALSE/NOT GIVEN statement..."
        // "SHORT_ANSWER" -> "Create a short answer question requiring ONE WORD..."
    ELSE:
        prefix = ""
    
    // Step 3: Format question type token
    type_token = f"<TYPE_{canonicalize(question_type)}>"
    // Example: "readingShortAnswer" -> "<TYPE_READINGSHORTANSWER>"
    
    // Step 4: Construct final source
    source = f"{irt_string} {prefix} {type_token}\nPassage: {passage}"
    
    RETURN source
END FUNCTION
