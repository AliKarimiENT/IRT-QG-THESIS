FUNCTION format_target_output(instruction, question, answer):
    /*
    Simple structured format WITHOUT control tokens
    This allows model to focus on content generation
    */
    
    target_parts = []
    
    IF instruction is not empty:
        target_parts.append(f"Instruction: {instruction.strip()}")
    
    IF question is not empty:
        target_parts.append(f"Question: {question.strip()}")
    
    IF answer is not empty:
        target_parts.append(f"Answer: {answer.strip()}")
    
    IF target_parts is empty:
        RAISE ValueError("Cannot create empty target")
    
    target = join(target_parts, separator="\n")
    
    RETURN target
END FUNCTION
