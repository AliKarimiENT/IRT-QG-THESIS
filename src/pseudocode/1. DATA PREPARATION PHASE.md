FUNCTION prepare_dataset(data_path, config):
    // Load and normalize data
    raw_data = load_table(data_path)  // Supports CSV/JSON/JSONL
    
    // Schema normalization
    column_mapping = {
        "PassageContext" -> "passage",
        "QuestionType" -> "question_type", 
        "InstructionContext" -> "ref_instruction",
        "Question" -> "ref_question",
        "Answer" -> "ref_answer"
    }
    data = rename_columns(raw_data, column_mapping)
    
    // Generate stable passage identifiers
    FOR each row in data:
        IF "passage_id" not exists:
            row["passage_id"] = MD5_hash(row["passage"])
    
    // Text normalization
    FOR each text_column in ["passage", "ref_instruction", "ref_question", "ref_answer"]:
        data[text_column] = normalize_special_tokens(data[text_column])
        // Converts: "BLANK" -> "<BLANK>", "SEP" -> "<SEP>"
    
    // Data validation and filtering
    data = filter_rows_where(
        passage_length >= 50 AND
        ref_instruction is not empty AND
        ref_question is not empty
    )
    
    RETURN data
END FUNCTION
